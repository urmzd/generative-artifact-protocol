    CREATE OR REPLACE VIEW <gap:target id="order-summary-view-name">order_summary</gap:target> AS
    SELECT
      <gap:target id="order-summary-order-id">o.id</gap:target> AS <gap:target id="order-summary-order-id-alias">order_id</gap:target>,
      <gap:target id="order-summary-order-number">o.order_number</gap:target> AS <gap:target id="order-summary-order-number-alias">order_number</gap:target>,
      <gap:target id="order-summary-user-id">o.user_id</gap:target>,
      <gap:target id="order-summary-user-name">u.first_name || ' ' || u.last_name</gap:target> AS <gap:target id="order-summary-user-name-alias">customer_name</gap:target>,
      <gap:target id="order-summary-status">o.status</gap:target>,
      <gap:target id="order-summary-subtotal">o.subtotal</gap:target>,
      <gap:target id="order-summary-tax">o.tax</gap:target>,
      <gap:target id="order-summary-shipping">o.shipping</gap:target>,
      <gap:target id="order-summary-discount">o.discount</gap:target>,
      <gap:target id="order-summary-total">o.total</gap:target>,
      <gap:target id="order-summary-item-count">COUNT(oi.id)</gap:target> AS <gap:target id="order-summary-item-count-alias">item_count</gap:target>,
      <gap:target id="order-summary-created-at">o.placed_at</gap:target> AS <gap:target id="order-summary-created-at-alias">placed_at</gap:target>
    FROM <gap:target id="order-summary-orders-table">orders</gap:target> o
    JOIN <gap:target id="order-summary-users-table">users</gap:target> u ON u.id = o.user_id
    LEFT JOIN <gap:target id="order-summary-order-items-table">order_items</gap:target> oi ON oi.order_id = o.id
    GROUP BY o.id, u.first_name, u.last_name;

    CREATE OR REPLACE VIEW <gap:target id="product-ratings-view-name">product_ratings</gap:target> AS
    SELECT
      <gap:target id="product-ratings-product-id">p.id</gap:target> AS <gap:target id="product-ratings-product-id-alias">product_id</gap:target>,
      <gap:target id="product-ratings-product-name">p.name</gap:target> AS <gap:target id="product-ratings-product-name-alias">product_name</gap:target>,
      <gap:target id="product-ratings-average-rating">COALESCE(AVG(r.rating), 0)</gap:target> AS <gap:target id="product-ratings-average-rating-alias">average_rating</gap:target>,
      <gap:target id="product-ratings-rating-count">COUNT(r.id)</gap:target> AS <gap:target id="product-ratings-rating-count-alias">rating_count</gap:target>
    FROM <gap:target id="product-ratings-products-table">products</gap:target> p
    LEFT JOIN <gap:target id="product-ratings-reviews-table">reviews</gap:target> r ON r.product_id = p.id
    GROUP BY p.id, p.name;

    CREATE OR REPLACE VIEW <gap:target id="low-stock-products-view-name">low_stock_products</gap:target> AS
    SELECT
      <gap:target id="low-stock-product-id">p.id</gap:target> AS <gap:target id="low-stock-product-id-alias">product_id</gap:target>,
      <gap:target id="low-stock-product-name">p.name</gap:target> AS <gap:target id="low-stock-product-name-alias">product_name</gap:target>,
      <gap:target id="low-stock-sku">p.sku</gap:target>,
      <gap:target id="low-stock-on-hand">i.quantity_on_hand</gap:target>,
      <gap:target id="low-stock-reserved">i.reserved_quantity</gap:target>,
      <gap:target id="low-stock-reorder-level">i.reorder_level</gap:target>
    FROM <gap:target id="low-stock-products-table">products</gap:target> p
    JOIN <gap:target id="low-stock-inventory-table">inventory</gap:target> i ON i.product_id = p.id
    WHERE <gap:target id="low-stock-condition">i.quantity_on_hand &lt;= i.reorder_level</gap:target>;
  </gap:target>
