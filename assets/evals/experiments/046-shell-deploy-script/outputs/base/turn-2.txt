#!/usr/bin/env bash
set -euo pipefail

IFS=$'\n\t'

DRY_RUN=0

APP_NAME="${APP_NAME:-my-node-app}"
SERVER_PORT="${SERVER_PORT:-22}"
DEPLOY_PATH="${DEPLOY_PATH:-/var/www/${APP_NAME}}"
RELEASES_DIR="${RELEASES_DIR:-${DEPLOY_PATH}/releases}"
SHARED_DIR="${SHARED_DIR:-${DEPLOY_PATH}/shared}"
CURRENT_LINK="${CURRENT_LINK:-${DEPLOY_PATH}/current}"
HEALTHCHECK_URL="${HEALTHCHECK_URL:-http://localhost:3000/health}"
NODE_VERSION="${NODE_VERSION:-18}"
BRANCH="${BRANCH:-main}"
PM2_APP_NAME="${PM2_APP_NAME:-${APP_NAME}}"
REMOTE_TARBALL_DIR="${REMOTE_TARBALL_DIR:-/tmp}"
LOCAL_BUILD_DIR="${LOCAL_BUILD_DIR:-./.deploy-build}"
TARBALL_NAME="${TARBALL_NAME:-${APP_NAME}-$(date +%Y%m%d%H%M%S).tar.gz}"
TARBALL_PATH="${TARBALL_PATH:-${LOCAL_BUILD_DIR}/${TARBALL_NAME}}"

SERVER_HOST="${SERVER_HOST:-}"
SERVERS=("${SERVERS[@]:-}")

DEPLOY_ENV_VARS=("${DEPLOY_ENV_VARS[@]:-}")

if [[ -z "${SERVER_HOST}" && ${#SERVERS[@]} -eq 0 ]]; then
  SERVERS=("${SERVER_HOST}")
fi

if [[ -n "${SERVER_HOST}" && ${#SERVERS[@]} -eq 0 ]]; then
  SERVERS=("${SERVER_HOST}")
fi

if [[ -n "${SERVER_HOST}" && ${#SERVERS[@]} -gt 0 ]]; then
  if [[ "${SERVERS[0]:-}" != "${SERVER_HOST}" ]]; then
    SERVERS=("${SERVER_HOST}" "${SERVERS[@]}")
  fi
fi

if [[ ${#SERVERS[@]} -eq 0 ]]; then
  SERVERS=("${SERVER_HOST}")
fi

if [[ "${SERVERS[0]:-}" == "" ]]; then
  SERVERS=()
fi

if [[ ${#SERVERS[@]} -eq 0 ]]; then
  SERVERS=("user@example.com")
fi

if [[ -t 1 ]]; then
  RED=$'\033[0;31m'
  GREEN=$'\033[0;32m'
  YELLOW=$'\033[1;33m'
  BLUE=$'\033[0;34m'
  RESET=$'\033[0m'
else
  RED=""
  GREEN=""
  YELLOW=""
  BLUE=""
  RESET=""
fi

log() { printf '%b\n' "${BLUE}==>${RESET} $*"; }
ok() { printf '%b\n' "${GREEN}✔${RESET} $*"; }
warn() { printf '%b\n' "${YELLOW}⚠${RESET} $*"; }
err() { printf '%b\n' "${RED}✖${RESET} $*" >&2; }

die() {
  err "$*"
  exit 1
}

trap 'err "Deployment failed at line $LINENO"; exit 1' ERR

print_cmd() { printf '%s\n' "$*"; }

run_or_print() {
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "$*"
  else
    eval "$*"
  fi
}

require_cmd() {
  for cmd in "$@"; do
    command -v "$cmd" >/dev/null 2>&1 || die "Required command not found: ${cmd}"
  done
}

cleanup_local() {
  [[ ${DRY_RUN} -eq 1 ]] && { print_cmd "rm -rf ${LOCAL_BUILD_DIR}"; return 0; }
  rm -rf "${LOCAL_BUILD_DIR}"
}

on_exit() {
  local code=$?
  [[ $code -eq 0 ]] && cleanup_local
}
trap on_exit EXIT

ssh_cmd() {
  local server="$1"
  local cmd="$2"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    printf 'ssh -p %q -o BatchMode=yes -o StrictHostKeyChecking=accept-new %q %q\n' "${SERVER_PORT}" "${server}" "bash -lc $(printf '%q' "$cmd")"
    return 0
  fi
  ssh -p "${SERVER_PORT}" -o BatchMode=yes -o StrictHostKeyChecking=accept-new "${server}" "bash -lc $(printf '%q' "$cmd")"
}

rsync_upload() {
  local server="$1"
  local src="$2"
  local dest="$3"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    printf 'rsync -az --delete -e %q %q %q\n' "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new" "${src}" "${server}:${dest}"
    return 0
  fi
  rsync -az --delete -e "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new" "${src}" "${server}:${dest}"
}

check_git_clean() {
  log "Checking git status..."
  git rev-parse --is-inside-work-tree >/dev/null 2>&1 || die "Not inside a git repository"
  [[ -z "$(git status --porcelain)" ]] || { git status --short; die "Working tree is not clean"; }
  ok "Git working tree is clean"
}

check_branch() {
  log "Checking current branch..."
  local current_branch
  current_branch="$(git rev-parse --abbrev-ref HEAD)"
  [[ "${current_branch}" == "${BRANCH}" ]] || die "Current branch '${current_branch}' does not match required branch '${BRANCH}'"
  ok "On branch ${BRANCH}"
}

run_tests() {
  log "Running tests..."
  [[ -f package.json ]] || die "package.json not found"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "npm test"
    ok "Tests would be run"
    return 0
  fi
  npm test
  ok "Tests passed"
}

check_ssh_all() {
  log "Checking SSH connectivity to all servers..."
  local server
  for server in "${SERVERS[@]}"; do
    if [[ ${DRY_RUN} -eq 1 ]]; then
      print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} echo connected"
    else
      ssh -p "${SERVER_PORT}" -o BatchMode=yes -o StrictHostKeyChecking=accept-new "${server}" "echo connected" >/dev/null
    fi
  done
  ok "SSH connectivity verified for all servers"
}

build_artifact() {
  log "Building application locally..."
  mkdir -p "${LOCAL_BUILD_DIR}"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "npm ci"
    print_cmd "npm run build"
    print_cmd "tar --exclude=.git --exclude=${LOCAL_BUILD_DIR} --exclude=node_modules --exclude=.env -czf ${TARBALL_PATH} ."
    ok "Build would complete and tarball would be created: ${TARBALL_PATH}"
    return 0
  fi
  npm ci
  npm run build
  tar --exclude=".git" --exclude="${LOCAL_BUILD_DIR}" --exclude="node_modules" --exclude=".env" -czf "${TARBALL_PATH}" .
  ok "Created tarball: ${TARBALL_PATH}"
}

prepare_remote_dirs() {
  local server="$1"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} mkdir -p ${RELEASES_DIR} ${SHARED_DIR}"
    return 0
  fi
  ssh_cmd "${server}" "mkdir -p '${RELEASES_DIR}' '${SHARED_DIR}'"
}

sync_env_file() {
  if [[ ${#DEPLOY_ENV_VARS[@]} -eq 0 ]]; then
    warn "No DEPLOY_ENV_VARS provided; skipping remote .env generation"
    return 0
  fi

  log "Uploading environment file to all servers..."
  local env_tmp
  env_tmp="$(mktemp)"
  {
    for kv in "${DEPLOY_ENV_VARS[@]}"; do
      printf '%s\n' "${kv}"
    done
  } > "${env_tmp}"

  local server
  for server in "${SERVERS[@]}"; do
    if [[ ${DRY_RUN} -eq 1 ]]; then
      print_cmd "scp -P ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${env_tmp} ${server}:${SHARED_DIR}/.env"
      print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} chmod 600 ${SHARED_DIR}/.env"
    else
      scp -P "${SERVER_PORT}" -o BatchMode=yes -o StrictHostKeyChecking=accept-new "${env_tmp}" "${server}:${SHARED_DIR}/.env"
      ssh_cmd "${server}" "chmod 600 '${SHARED_DIR}/.env'"
    fi
  done

  rm -f "${env_tmp}"
  ok "Environment file uploaded to all servers"
}

upload_tarballs_parallel() {
  log "Uploading tarball to all servers in parallel..."
  local server
  local pids=()
  for server in "${SERVERS[@]}"; do
    if [[ ${DRY_RUN} -eq 1 ]]; then
      print_cmd "rsync -az --delete -e 'ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new' ${TARBALL_PATH} ${server}:${REMOTE_TARBALL_DIR}/${TARBALL_NAME}"
    else
      rsync -az -e "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new" "${TARBALL_PATH}" "${server}:${REMOTE_TARBALL_DIR}/${TARBALL_NAME}" &
      pids+=($!)
    fi
  done

  if [[ ${DRY_RUN} -eq 0 ]]; then
    local pid
    for pid in "${pids[@]}"; do
      wait "${pid}"
    done
  fi
  ok "Tarball upload completed for all servers"
}

extract_release() {
  local server="$1"
  local release_dir="${RELEASES_DIR}/$(date +%Y%m%d%H%M%S)-${APP_NAME}"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} \"mkdir -p '${release_dir}' && tar -xzf '${REMOTE_TARBALL_DIR}/${TARBALL_NAME}' -C '${release_dir}' && ln -sfn '${SHARED_DIR}/.env' '${release_dir}/.env' || true\""
    printf '%s\n' "${release_dir}"
    return 0
  fi
  ssh_cmd "${server}" "mkdir -p '${release_dir}' && tar -xzf '${REMOTE_TARBALL_DIR}/${TARBALL_NAME}' -C '${release_dir}' && ln -sfn '${SHARED_DIR}/.env' '${release_dir}/.env' || true && echo '${release_dir}'" | tail -n 1
}

install_and_migrate() {
  local server="$1"
  local release_dir="$2"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} \"cd '${release_dir}' && export NODE_ENV=production && npm ci --omit=dev && (npm run migrate || npm run db:migrate || true)\""
    return 0
  fi
  ssh_cmd "${server}" "
    set -euo pipefail
    cd '${release_dir}'
    export NODE_ENV=production
    if command -v nvm >/dev/null 2>&1; then
      . \"\$HOME/.nvm/nvm.sh\"
      nvm use '${NODE_VERSION}' >/dev/null
    fi
    npm ci --omit=dev
    if npm run | grep -qE '(^| )migrate( |$)'; then
      npm run migrate
    elif npm run | grep -qE '(^| )db:migrate( |$)'; then
      npm run db:migrate
    else
      echo 'No migration script found; skipping'
    fi
  "
}

swap_symlink_and_restart() {
  local server="$1"
  local release_dir="$2"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} \"cd '${DEPLOY_PATH}' && ln -sfn '${release_dir}' current && pm2 startOrReload ecosystem.config.js --env production --update-env || pm2 reload '${PM2_APP_NAME}' --update-env || pm2 restart '${PM2_APP_NAME}' --update-env\""
    return 0
  fi
  ssh_cmd "${server}" "
    set -euo pipefail
    cd '${DEPLOY_PATH}'
    ln -sfn '${release_dir}' current
    if command -v pm2 >/dev/null 2>&1; then
      cd current
      pm2 startOrReload ecosystem.config.js --env production --update-env || pm2 reload '${PM2_APP_NAME}' --update-env || pm2 restart '${PM2_APP_NAME}' --update-env
    else
      echo 'PM2 not installed on remote server' >&2
      exit 1
    fi
  "
}

health_check() {
  local server="$1"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} curl -fsS '${HEALTHCHECK_URL}'"
    return 0
  fi
  ssh -p "${SERVER_PORT}" -o BatchMode=yes -o StrictHostKeyChecking=accept-new "${server}" "curl -fsS '${HEALTHCHECK_URL}'" >/dev/null
}

tail_logs() {
  local server="$1"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} \"pm2 logs '${PM2_APP_NAME}' --lines 50 --nostream\""
    return 0
  fi
  ssh_cmd "${server}" "pm2 logs '${PM2_APP_NAME}' --lines 50 --nostream" || true
}

rollback_server() {
  local server="$1"
  local previous_release="$2"
  if [[ ${DRY_RUN} -eq 1 ]]; then
    print_cmd "ssh -p ${SERVER_PORT} -o BatchMode=yes -o StrictHostKeyChecking=accept-new ${server} \"cd '${DEPLOY_PATH}' && ln -sfn '${previous_release}' current && pm2 reload '${PM2_APP_NAME}' --update-env || pm2 restart '${PM2_APP_NAME}' --update-env || true\""
    return 0
  fi
  ssh_cmd "${server}" "
    set -euo pipefail
    cd '${DEPLOY_PATH}'
    ln -sfn '${previous_release}' current
    if command -v pm2 >/dev/null 2>&1; then
      pm2 reload '${PM2_APP_NAME}' --update-env || pm2 restart '${PM2_APP_NAME}' --update-env || true
    fi
  " || true
}

deploy_server() {
  local server="$1"
  local release_dir previous_release

  prepare_remote_dirs "${server}"
  release_dir="$(extract_release "${server}")"
  install_and_migrate "${server}" "${release_dir}"

  if [[ ${DRY_RUN} -eq 1 ]]; then
    previous_release="${DEPLOY_PATH}/previous-release"
  else
    previous_release="$(ssh_cmd "${server}" "readlink -f '${CURRENT_LINK}' 2>/dev/null || true")"
  fi

  swap_symlink_and_restart "${server}" "${release_dir}"

  if health_check "${server}"; then
    ok "Deployment successful on ${server}"
    tail_logs "${server}"
    return 0
  fi

  err "Post-deploy verification failed on ${server}"
  [[ -n "${previous_release}" ]] && rollback_server "${server}" "${previous_release}"
  tail_logs "${server}"
  return 1
}

check_git_clean
check_branch
run_tests

check_ssh_all
build_artifact
sync_env_file
upload_tarballs_parallel

failures=0
for server in "${SERVERS[@]}"; do
  if ! deploy_server "${server}"; then
    failures=$((failures + 1))
  fi
done

if [[ ${failures} -ne 0 ]]; then
  die "Deployment failed on ${failures} server(s)"
fi

ok "Deployment completed successfully on all servers"
