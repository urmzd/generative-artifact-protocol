<gap:target id="helpers">
log() { echo -e "${BLUE}[INFO]${NC} $*"; }
success() { echo -e "${GREEN}[OK]${NC} $*"; }
warn() { echo -e "${YELLOW}[WARN]${NC} $*"; }
fail() { echo -e "${RED}[ERROR]${NC} $*"; }

rollback() {
  warn "Rollback triggered"
  ssh "${SERVER_USER}@${SERVER_HOST}" "
    set -e
    if [ -L '${REMOTE_CURRENT_LINK}' ] && [ -f '${REMOTE_RELEASES_DIR}/previous_release' ]; then
      PREV_RELEASE=\$(cat '${REMOTE_RELEASES_DIR}/previous_release')
      ln -sfn \"${REMOTE_RELEASES_DIR}/\${PREV_RELEASE}\" '${REMOTE_CURRENT_LINK}'
      pm2 restart '${PM2_APP_NAME}' || true
      echo 'Rolled back to previous release: '\$PREV_RELEASE
    else
      echo 'No rollback target available'
      exit 1
    fi
  "
}

trap 'fail "Deployment failed"; rollback; exit 1' ERR
</gap:target>

<gap:target id="pre-deploy-checks">log "Verifying SSH connectivity"
run_cmd 'ssh -o BatchMode=yes -o ConnectTimeout=10 "${SERVER_USER}@${SERVER_HOST}" "echo connected"'</gap:target>

<gap:target id="build">log "Installing dependencies"
run_cmd 'npm ci'

log "Running build"
run_cmd 'npm run build'

log "Creating tarball"
run_cmd 'tar -czf "${LOCAL_ARTIFACT_PATH}" "${BUILD_DIR}" package.json package-lock.json'</gap:target>

<gap:target id="deploy">log "Preparing remote directories"
run_cmd 'for SERVER in "${SERVERS[@]}"; do ssh "${SERVER_USER}@${SERVER}" "set -e; mkdir -p '\''${REMOTE_RELEASES_DIR}'\'' '\''${REMOTE_SHARED_DIR}'\''"; done'

log "Uploading artifact to all servers in parallel"
run_cmd 'pids=(); for SERVER in "${SERVERS[@]}"; do rsync -az --delete "${LOCAL_ARTIFACT_PATH}" "${SERVER_USER}@${SERVER}:${REMOTE_ARTIFACT_PATH}" & pids+=("$!"); done; for pid in "${pids[@]}"; do wait "$pid"; done'

log "Deploying on remote servers"
run_cmd 'for SERVER in "${SERVERS[@]}"; do ssh "${SERVER_USER}@${SERVER}" "set -euo pipefail; RELEASE_NAME=\$(basename '\''${ARTIFACT_NAME}'\'' .tar.gz); RELEASE_DIR='\''${REMOTE_RELEASES_DIR}/'\''\${RELEASE_NAME}; mkdir -p \"$RELEASE_DIR\"; tar -xzf '\''${REMOTE_ARTIFACT_PATH}'\'' -C \"$RELEASE_DIR\"; rm -f '\''${REMOTE_ARTIFACT_PATH}'\''; cd \"$RELEASE_DIR\"; npm ci --omit=dev; cat > .env <<'"'"'EOF'"'"'; NODE_ENV=production; PORT=3000; LOG_LEVEL=info; EOF; if [ -f '\''${REMOTE_RELEASES_DIR}/current_release'\'' ]; then cp '\''${REMOTE_RELEASES_DIR}/current_release'\'' '\''${REMOTE_RELEASES_DIR}/previous_release'\''; fi; echo \"$RELEASE_NAME\" > '\''${REMOTE_RELEASES_DIR}/current_release'\''; if [ -f migrations.sh ]; then chmod +x migrations.sh; ./migrations.sh; elif npm run | grep -q '\''migrate'\''; then npm run migrate; fi; ln -sfn \"$RELEASE_DIR\" '\''${REMOTE_CURRENT_LINK}'\''; pm2 restart '\''${PM2_APP_NAME}'\'' --update-env || pm2 start ecosystem.config.js --only '\''${PM2_APP_NAME}'\'' --update-env; pm2 save"; done'</gap:target>

<gap:target id="post-deploy-verification">log "Waiting for service to stabilize"
run_cmd 'sleep 5'

log "Checking health endpoint"
run_cmd 'curl -fsS --max-time 10 "${HEALTHCHECK_URL}" >/dev/null'

log "Showing recent logs"
run_cmd 'ssh "${SERVER_USER}@${SERVER_HOST}" "pm2 logs '\''${PM2_APP_NAME}'\'' --lines 50 --nostream"'</gap:target>

DRY_RUN=false

run_cmd() {
  if [[ "$DRY_RUN" == true ]]; then
    printf '%s\n' "$*"
  else
    eval "$@"
  fi
}while [[ $# -gt 0 ]]; do
  case "$1" in
    --dry-run)
      DRY_RUN=true
      shift
      ;;
    *)
      echo "Unknown argument: $1" >&2
      exit 1
      ;;
  esac
done

if [[ "$DRY_RUN" == true ]]; then
  warn "Dry-run mode enabled: commands will be printed only"
fi</gap:target>
